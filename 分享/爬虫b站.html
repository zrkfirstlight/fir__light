# 导入requests和BeautifulSoup库，用于发送HTTP请求和解析HTML内容。
import requests
from bs4 import BeautifulSoup

# 定义了一个User-Agent，以模拟正常的用户请求。
headers = {
    "User-Agent": "自己给"}
# 使用requests库发送GET请求到指定URL
response = requests.get(f"想要的论文的网址", headers=headers)
# 打印HTTP响应状态码，检查请求是否成功。
print(response.status_code)
# 将获取到的网页内容以字符串形式存储在变量html中。
html = response.text
# print(html)
# 使用BeautifulSoup库将网页内容解析为一个可遍历的对象。
soup = BeautifulSoup(html, "html.parser")
# 观察网页结构，找到容器标签，将其信息全部提取并打印
image_urls = []
all_div = soup.findAll("div", attrs={"style": "padding: 0 4px;"})
for div in all_div:
    images = div.find_all('img')
    for image in images:
        print(f"src={image.get('src')}, alt={image.get('alt')}")
        image_urls.append(image.get('src'))

##
# todo: 设定文件保存名字，
all_file_name = "自己喜欢的名字"

##
# todo: 单纯保存图片

image_num = 0
for image_url in image_urls:
    # 发送HTTP请求获取图片数据
    response = requests.get(image_url)
    image_num += 1
    try:
        if response.status_code == 200:
            # 提取文件名
            filename = f"图片合集/论文图片{image_num}.png"
            # 保存图片到本地
            with open(filename, 'wb') as file:
                file.write(response.content)
            print(f"图片已成功下载到 {filename}")
        else:
            print("无法下载图片，错误状态码:", response.status_code)
    except:
        print("当前目录下没有文件夹 图片合集 ，无法保存图片")

# todo:下载图片并保存成PDF形式
from reportlab.lib.pagesizes import letter
from reportlab.pdfgen import canvas

# 创建PDF文档
pdf_filename = f"{all_file_name}.pdf"
c = canvas.Canvas(pdf_filename, pagesize=letter)

# 设置页面尺寸
width, height = letter

# 逐个遍历图片URL，并绘制到PDF中不同页
for image_url in image_urls:
    c.drawImage(image_url, 0, 0, width, height)  # 将图片绘制到整个页面
    c.showPage()

# 关闭PDF文档
c.save()

print(f"图片已成功保存为 {pdf_filename}")
